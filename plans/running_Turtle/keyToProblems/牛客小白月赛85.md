#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef double db;

void solve(){
    ll n, m;
    cin>>n>>m;
    vector<ll> a(n+5);
    vector<ll> s(n+5);
    for(int i=1; i<=n; i++) cin>>a[i];
    for(int i=n; i>=1; i--) s[i]=s[i+1]+a[i];
    ll cur=m, tot=0;
    for(int i=1; i<=n; i++){
        if(a[i]==0) tot+=m-cur, cur=m;
        else{
            cur-=a[i];
            if(cur<=0){
                cout<<"NO";
                return;
            }
        }
        if(cur>s[i+1]){
            cout<<tot+n-1;
            return ;
        }
    }
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);

    int T = 1;
    // cin >> T;
    while(T --){
        solve();
    }    

    return 0;
}
